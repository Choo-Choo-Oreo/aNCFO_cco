 country_event = {
	id = exhaustion.101
	title = exhaustion.101.t
	desc = exhaustion.101.d
	
	picture = GFX_report_event_german_troops_paris

	is_triggered_only = yes

	option = {
		name = exhaustion.101.a
		ai_chance = {factor = 100}
		if = {
			limit = { 
				NOT = { has_country_flag = original_capital_lost }
				has_variable = war_exhaustion
			}
			custom_effect_tooltip = raise_by_50
			add_to_variable = {war_exhaustion = 50}
			log = "50 kms"
			event_target:capital_taker = {
				country_event = {
					id = exhaustion.104
					days = 1
					random_hours = 12 
				}
			}
			clamp_variable = {
				var = war_exhaustion
				min = 0
				max = 100
			}
		}
		else_if = {
			limit = {
				has_variable = war_exhaustion
			}
			custom_effect_tooltip = raise_by_20
			add_to_variable = {war_exhaustion = 20}
			log = "20 kms"
			clamp_variable = {
				var = war_exhaustion
				min = 0
				max = 100
			}
		}
		
		
		
	}
 }
