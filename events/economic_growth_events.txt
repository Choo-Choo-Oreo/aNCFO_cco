add_namespace = economic_growth



# Capitalist Industrial Growth
country_event = {
	id = economic_growth.1
	title = economic_growth.1.t
	desc = economic_growth.1.d
	picture = GFX_capitalist_buisness

	is_triggered_only = yes

	option = {
		name = economic_growth.1.a
		random_owned_controlled_state = {
			limit = { 
				free_building_slots = {
					building = industrial_complex
					size > 0
					include_locked = yes
				}
			}
			add_extra_state_shared_building_slots = 1
			add_building_construction = {
				type = industrial_complex
				level = 1
				instant_build = yes
			}
		}
	}

}

# NonColonial Neutral Economic Growth
country_event = {
	id = economic_growth.21

	hidden = yes

	trigger = {
		AND = {
			has_government = authoritarianism
			OR = { 
				has_idea = civilian_economy
				has_idea = low_economic_mobilisation
				has_idea = undisturbed_isolation
				has_idea = isolation
			}
			any_state = {
				is_controlled_by = ROOT
				is_owned_by = ROOT
			}
		}
	}

	mean_time_to_happen = {
		days = 60
		modifier = {
			factor = 1
			has_idea = civilian_economy
		}
		modifier = {
			factor = 1.5
			has_idea = low_economic_mobilisation
		}
		modifier = {
			factor = 1.5
			has_idea = isolation
		}
		modifier = {
			factor = 2
			has_idea = undisturbed_isolation
		}
	}

	immediate = {
		fixed_random_seed = no
		random_list = {
			40 = { # Neutral Infastructure Growth
				country_event = economic_growth.17
			}
			30 = { # Neutral Regional Growth
				country_event = economic_growth.18
			}
			20 = { # Neutral Industrial Growth
				country_event = economic_growth.16
			}
			10 = { # Neutral National Monument
				country_event = economic_growth.20
			}
		}
	}

}
