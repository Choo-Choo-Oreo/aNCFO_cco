FAI_update_colonial_empire = {
	set_temp_variable = { non_connected_states_count = 0 }
	# Iterate through every state owned by the country
	every_owned_state = {
		limit = {
			# Only count states not connected to the capital by land
			NOt = { is_in_home_area = yes impassable = yes }
		}
		# Increment the country's temp variable for each found state
		add_to_temp_variable = { ROOT.non_connected_states_count = 1 }
	}
	# If the count exceeds 6, mark them as a colonial empire
	IF = {
		limit = { check_variable = { non_connected_states_count > 6 } }
		set_variable = { is_colonial_empire = non_connected_states_count }
		IF = { LIMIT = { is_debug = yes } log = "[GetDateText]: [THIS.GetName]: Non-Connected States Count [?is_colonial_empire] - is_colonial_empire = yes" }
	}
}

FAI_update_primarily_coastal = {
	set_temp_variable = { FAI_state_coastal_counting = 0 }
	every_owned_state = {
		limit = {
			is_core_of = ROOT
			is_coastal = yes
		}
		# Increment the temporary counter for every state that fails the check
		add_to_temp_variable = { ROOT.FAI_state_coastal_counting = 1 }
	}
	IF = { LIMIT = { is_debug = yes } log = "[GetDateText]: [THIS.GetName]: Coastal States Count [?FAI_state_coastal_counting]" }
	set_variable = { FAI_state_coastal_count = FAI_state_coastal_counting }

	set_temp_variable = { FAI_state_non_coastal_counting = 0 }
	every_owned_state = {
		limit = {
			is_core_of = ROOT
			NOT = { is_coastal = yes }
		}
		# Increment the temporary counter for every state that fails the check
		add_to_temp_variable = { ROOT.FAI_state_non_coastal_counting = 1 }
	}
	IF = { LIMIT = { is_debug = yes } log = "[GetDateText]: [THIS.GetName]: Non-Coastal States Count [?FAI_state_non_coastal_counting]" }
	set_variable = { FAI_state_non_coastal_count = FAI_state_non_coastal_counting }
}

FAI_update_no_connected_neighbors = {
	# Default to 1 (Assuming we have no neighbors)
	set_variable = { FAI_no_connected_neighbors = 1 }

	# Check if we DO have a neighbor connected to the capital
	IF = {
		limit = {
			# Returns true if the country has NO land neighbors connected to the capital
			NOT = {
				any_owned_state = {
					is_in_home_area = yes
					any_neighbor_state = {
						NOT = { is_owned_by = ROOT }
					}
				}
			}
		}
		IF = { LIMIT = { is_debug = yes } log = "[GetDateText]: [THIS.GetName]: has_no_connected_neighbors = yes" }
		set_variable = { FAI_no_connected_neighbors = 1 }
	}
}

FAI_calculate_oil_status = {
	# 1. Grab base values
	set_temp_variable = { t_net_balance = resource@oil }
	set_temp_variable = { t_imports = resource_imported@oil }

	# 2. Calculate Domestic Balance (Net minus Imports)
	set_temp_variable = { t_domestic_balance = t_net_balance }
	subtract_from_temp_variable = { t_domestic_balance = t_imports }

	# 3. Determine the 3 States
	IF = {
		# STATE 1: Deficit (Net balance is below 0)
		LIMIT = { 
			check_variable = { t_net_balance < 0 } 
		}
		set_variable = { oil_deficit = t_net_balance }
		multiply_variable = { oil_deficit = -1 } # Make it a positive number for your records
		set_variable = { oil_surplus = 0 }
		set_variable = { oil_status = 1 } # Flag 1 = Deficit
		
		IF = { LIMIT = { is_debug = yes } log = "  Oil Status: DEFICIT of [?oil_deficit]" }
	}
	ELSE_IF = {
		# STATE 2: Surplus w/o Trade (Domestic balance is 0 or higher)
		LIMIT = { 
			check_variable = { var = t_domestic_balance value = 0 compare = greater_than_or_equals } 
		}
		set_variable = { oil_surplus = t_net_balance }
		set_variable = { oil_deficit = 0 }
		set_variable = { oil_status = 2 } # Flag 2 = Self-Sufficient Surplus
		
		IF = { LIMIT = { is_debug = yes } log = "  Oil Status: SURPLUS W/O TRADE of [?oil_surplus]" }
	}
	ELSE = {
		# STATE 3: Surplus w/ Trade (Net is >= 0, but you rely on imports to achieve it)
		set_variable = { oil_surplus = t_net_balance }
		set_variable = { oil_deficit = 0 }
		set_variable = { oil_status = 3 } # Flag 3 = Trade-Reliant Surplus
		
		IF = { LIMIT = { is_debug = yes } log = "  Oil Status: SURPLUS W/ TRADE of [?oil_surplus]" }
	}
}

FAI_deployed_total = {
	# Compute on_monthly_deployed_total = army + navy + airforce
	set_temp_variable = { on_monthly_deployed_total = deployed_army_manpower_k }
	add_to_temp_variable = { on_monthly_deployed_total = deployed_navy_manpower_k }
	add_to_temp_variable = { on_monthly_deployed_total = deployed_airforce_manpower_k }
	# Compute on_monthly_reserved = manpower_k - on_monthly_deployed_total
	set_temp_variable = { on_monthly_reserved = manpower_k }
	subtract_from_temp_variable = { on_monthly_reserved = on_monthly_deployed_total }
	set_variable = { var = deployed_total value = on_monthly_deployed_total }
	set_variable = { var = reserved value = on_monthly_reserved }
	# Used for "check_variable = { var = reserved value = deployed_total compare = greater_than }"
	IF = { LIMIT = { is_debug = yes } log = "Deployed Total: [?deployed_total] | Reserved Manpower: [?reserved]" }

	# Store deployed fighters in temp variable first
	set_temp_variable = { deployed_fighters = num_deployed_planes_with_type@fighter }
	set_temp_variable = { stockpile_planes = num_equipment@small_plane_airframe }
	IF = { LIMIT = { is_debug = yes } log = "Deployed Fighters: [?deployed_fighters] | Stockpile Planes: [?stockpile_planes]" }
}

	set_variable = { var = FAI_convoy_target value = deployed_total }

	IF = { # Always have 1k convoys as major
		limit = {
			#ROOT = { is_major = yes }
			check_variable = { var = FAI_convoy_target value = 1000 compare = less_than }

	}

	multiply_variable = { var = FAI_convoy_of2_limit value = 1.2 }
	IF = { LIMIT = { is_debug = yes } log = "Convoy Target: [?FAI_convoy_target] | OF1: [?FAI_convoy_of1_limit] | OF2: [?FAI_convoy_of2_limit]" }
}

FAI_calc_queue_batch_size = {
	set_temp_variable = { FAI_batch_limit = num_of_civilian_factories }
	divide_temp_variable = { FAI_batch_limit = 30 }
	multiply_temp_variable = { FAI_batch_limit = FAI_queue_base }
	clamp_temp_variable = { var = FAI_batch_limit min = 1 }
	round_temp_variable = FAI_batch_limit
}

FAI_electrical_grid_queuing = {
	set_temp_variable = { FAI_queue_base = 0.5 }
	FAI_calc_queue_batch_size = yes
	IF = {
		LIMIT = { NOT = { has_country_flag = cbq_infrastructure } }
		meta_effect = {
			text = {
				every_core_state = { # "Reinforced Electrical Grid"/"High Capacity Electrical Grid" Queuing
					random_select_amount = [BATCH_SIZE] # "random_select_amount" prevents "energy_infrastructure"/"industrial_infrastructure" Queue overload
					LIMIT = {
						is_core_of = ROOT
						check_variable = { var = building_level@industrial_infrastructure value = 1 compare = less_than }
						check_variable = { var = building_level@energy_infrastructure	  value = 1 compare = less_than }
					}
					set_temp_variable = { state_factory_count = building_level@industrial_complex }
					add_to_temp_variable = { state_factory_count = building_level@arms_factory }
					add_to_temp_variable = { state_factory_count = building_level@dockyard }
					IF = { # industrial_infrastructure
						LIMIT = {
							free_building_slots = { building = industrial_infrastructure size > 0 }
							OR = {
								check_variable = { var = resource@oil		value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@coal		value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@aluminium	value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@rubber	value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@tungsten	value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@steel		value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@chromium	value = 20 compare = greater_than_or_equals }
							}
						}
						add_building_construction = { type = industrial_infrastructure level = 1 instant_build = no }
						log = "[ROOT_GNAME]: SUCCESS - Queued industrial_infrastructure in [This.GetName]. Setting Flag."
						owner = { set_country_flag = { flag = cbq_infrastructure days = 180 } }
					}
					ELSE_IF = { # energy_infrastructure
						LIMIT = {
							free_building_slots = { building = energy_infrastructure size > 0 }
							OR = {
								AND = {
									owner = { check_variable = { var = num_of_factories value = 100 compare = greater_than_or_equals } }
									check_variable = { var = num_of_factories value = 2 compare = greater_than_or_equals }
								}
								AND = {
									owner = { check_variable = { var = num_of_factories value = 100 compare = less_than_or_equals } }
									check_variable = { var = num_of_factories value = 9 compare = greater_than_or_equals }
								}
							}
						}
						add_building_construction = { type = energy_infrastructure level = 1 instant_build = no }
						log = "[ROOT_GNAME]: SUCCESS - Queued energy_infrastructure in [This.GetName]. Setting Flag."
						owner = { set_country_flag = { flag = cbq_infrastructure days = 180 } }
					}
				}
				log = "[ROOT_GNAME]: Electrical Grid Queuing Batch Size: [BATCH_SIZE]"
			}
			ROOT_GNAME = "[ROOT.GetName]"
			BATCH_SIZE = "[?FAI_batch_limit|.0]"
		}
	}
	ELSE = {
		log = "[ROOT.GetName]: Electrical Grid Cooldown Active - Script Skipped."
	}
}
