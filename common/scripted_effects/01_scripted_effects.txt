# Important to put HIGHEST first
unitism_revolutionary_warfare = {
	if = {
		limit = {
			not = {
				has_idea = revolutionary_warfare3
				has_idea = revolutionary_warfare2
				has_idea = revolutionary_warfare1
				has_idea = revolutionary_warfare0
			}
		}
		add_ideas = revolutionary_warfare3
	}
	else_if = {
		limit = { has_idea = revolutionary_warfare3 }
		swap_ideas = {
			remove_idea = revolutionary_warfare3
			add_idea = revolutionary_warfare2
		}
	}
	else_if = {
		limit = { has_idea = revolutionary_warfare2 }
		swap_ideas = {
			remove_idea = revolutionary_warfare2
			add_idea = revolutionary_warfare1
		}
	}
	else_if = {
		limit = { has_idea = revolutionary_warfare1 }
		swap_ideas = {
			remove_idea = revolutionary_warfare1
			add_idea = revolutionary_warfare0
		}
	}
}
