#Important to put LOWEST first
MRC_center_of_knowledge = {
	if = {
		limit = { has_idea = center_of_knowledge1 }
		custom_effect_tooltip = HHE_center_of_knowledge_tt
		swap_ideas = {
			remove_idea = center_of_knowledge1
			add_idea = center_of_knowledge0
		}
	}
	if = {
		limit = { NOT = { has_idea = center_of_knowledge1 } }
		custom_effect_tooltip = HHE_center_of_knowledge_tt
		add_ideas = center_of_knowledge1
	}
}
