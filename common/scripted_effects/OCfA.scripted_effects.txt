OCfA_calculate_battalion_capacity_linear = {
	# --- LINEAR SCALING ---
	set_temp_variable = { var = OCfA_curve_value value = OCfA_capacity_diff }
	divide_temp_variable = { var = OCfA_curve_value value = @OCfA_linear_divisor } 

	# --- RESTORE DIRECTION ---
	IF = {
		LIMIT = { 
			check_variable = { var = OCfA_capacity_diff value = 0 compare = less_than } 
			check_variable = { var = OCfA_curve_value value = 0 compare = greater_than } 
		}
		multiply_temp_variable = { var = OCfA_curve_value value = -1 }
	}
}

OCfA_calculate_battalion_capacity_formula = {
	# --- INITIALIZE BASE VARIABLE ---
	set_variable = { var = OCfA_base_units value = 0 }

	# --- STEP 1: POPULATION & STATES CALCULATION ---
	# Calculates weighted population and adds to base units
	set_variable = { var = OCfA_count_core value = 0 }
	set_variable = { var = OCfA_pop_k_core value = 0 }
	set_variable = { var = OCfA_count_non_core value = 0 }
	set_variable = { var = OCfA_pop_k_non_core value = 0 }
	every_owned_state = {
		IF = {
			LIMIT = { is_core_of = ROOT }
			add_to_variable = { var = ROOT.OCfA_count_core value = 1 }
			add_to_variable = { var = ROOT.OCfA_pop_k_core value = state_population_k }
			#log = "[Root.GetTag] CORE STATE: [This.GetName]"
		}
		ELSE = {
			add_to_variable = { var = ROOT.OCfA_count_non_core value = 1 }
			add_to_variable = { var = ROOT.OCfA_pop_k_non_core value = state_population_k }
			#log = "[Root.GetTag] NON-CORE STATE: [This.GetName]"
		}
	}
	# FOR SOME REASON EDITING CAUSES MASSIVE ISSUES LIVE DURING GAMEPLAY, YOU MUST RESTART THE GAME WHEN EDITING ANY OF THE CODE!!!
	set_variable = { var = OCfA_core_pop value = OCfA_pop_k_core }
	multiply_variable = { var = OCfA_core_pop value = @OCfA_pop_core_weight }
	set_variable = { var = OCfA_core_mult value = 1.0 }
	add_to_variable = { var = OCfA_core_mult value = ROOT.modifier@OCfA_battalion_limit_pop_core_bonus }
	multiply_variable = { var = OCfA_core_pop value = OCfA_core_mult }
	divide_variable = { var = OCfA_core_pop value = @OCfA_pop_divisor }
	add_to_variable = { var = OCfA_base_units value = OCfA_core_pop }

	set_variable = { var = OCfA_non_core_pop value = OCfA_pop_k_non_core }
	multiply_variable = { var = OCfA_non_core_pop value = @OCfA_pop_non_core_weight }
	set_variable = { var = OCfA_non_core_mult value = 1.0 }
	add_to_variable = { var = OCfA_non_core_mult value = ROOT.modifier@OCfA_battalion_limit_pop_non_core_bonus }
	multiply_variable = { var = OCfA_non_core_pop value = OCfA_non_core_mult }
	divide_variable = { var = OCfA_non_core_pop value = @OCfA_pop_divisor }
	add_to_variable = { var = OCfA_base_units value = OCfA_non_core_pop }

	# Apply global population modifiers
	set_temp_variable = { var = OCfA_global_pop_mult value = 1.0 }
	add_to_temp_variable = { var = OCfA_global_pop_mult value = modifier@OCfA_battalion_limit_pop_bonus }
	multiply_variable = { var = OCfA_base_units value = OCfA_global_pop_mult }

	# Calculates weighted state count and adds to base units
	set_variable = { var = OCfA_base_states value = num_owned_states }
	multiply_variable = { var = OCfA_base_states value = @OCfA_state_weight }
	add_to_variable = { var = OCfA_base_units value = OCfA_base_states } 

	# --- STEP 2: FACTORY COUNT CALCULATION ---
	# Calculates weighted factories and applies the Industry Factor Ratio
	set_variable = { var = OCfA_factory_points value = 0 } 

	set_variable = { var = OCfA_mil_factories value = num_of_military_factories } 
	multiply_variable = { var = OCfA_mil_factories value = @OCfA_mil_weight } 
	add_to_variable = { var = OCfA_factory_points value = OCfA_mil_factories } 

	set_variable = { var = OCfA_civ_factories value = num_of_civilian_factories } 
	multiply_variable = { var = OCfA_civ_factories value = @OCfA_civ_weight } 
	add_to_variable = { var = OCfA_factory_points value = OCfA_civ_factories } 

	set_variable = { var = OCfA_nav_dockyards value = num_of_naval_factories } 
	multiply_variable = { var = OCfA_nav_dockyards value = @OCfA_dockyard_weight } 
	add_to_variable = { var = OCfA_factory_points value = OCfA_nav_dockyards }

	set_temp_variable = { var = OCfA_factory_mult value = 1.0 }
	add_to_temp_variable = { var = OCfA_factory_mult value = modifier@OCfA_battalion_limit_factory_bonus }
	multiply_variable = { var = OCfA_factory_points value = OCfA_factory_mult }

	# Industry Factor Ratio (Scaling)
	set_temp_variable = { var = OCfA_divisor value = OCfA_factory_points }
	add_to_temp_variable = { var = OCfA_divisor value = OCfA_base_units }

	set_variable = { var = OCfA_industry_factor value = OCfA_factory_points }
	subtract_from_variable = { var = OCfA_industry_factor value = OCfA_base_units }
	divide_variable = { var = OCfA_industry_factor value = OCfA_divisor }
	add_to_variable = { var = OCfA_industry_factor value = 1 }
	
	multiply_variable = { var = OCfA_industry_factor value = @industry_scaling_multiplier }
	clamp_variable = { var = OCfA_industry_factor min = @min_industry_factor }

	# Finalize Industry Scaling to Capacity
	set_variable = { var = OCfA_calculated_battalion_capacity value = OCfA_base_units }
	multiply_variable = { var = OCfA_calculated_battalion_capacity value = OCfA_industry_factor }

	# --- STEP 3: ECONOMIC GAMERULE MULTIPLIERS ---
	# Economic Mobilization Multiplier
	set_variable = { var = OCfA_econ_efficiency_factor value = 1.0 }
	IF = { 
		LIMIT = { has_game_rule = { rule = OCfA_mobilization_scaling_rule option = MOB_ON } }
		set_variable = { var = OCfA_current_cg_total value = modifier@consumer_goods_expected_value }
		set_temp_variable = { var = OCfA_cg_multiplier value = 1.0 }
		add_to_temp_variable = { var = OCfA_cg_multiplier value = modifier@consumer_goods_factor }
		multiply_variable = { var = OCfA_current_cg_total value = OCfA_cg_multiplier }
		subtract_from_variable = { var = OCfA_econ_efficiency_factor value = OCfA_current_cg_total }
		clamp_variable = { var = OCfA_econ_efficiency_factor min = 0.0 max = 1.0 }
		multiply_variable = { var = OCfA_calculated_battalion_capacity value = OCfA_econ_efficiency_factor }
	}

	# Infrastructure Strain Multiplier
	IF = {
		LIMIT = { has_game_rule = { rule = OCfA_strain_rule option = STRAIN_ON } }
		set_variable = { var = OCfA_infra_strain value = num_of_military_factories }
		multiply_variable = { var = OCfA_infra_strain value = @OCfA_upkeep_per_military }
		multiply_variable = { var = OCfA_infra_strain value = OCfA_econ_efficiency_factor }
		set_temp_variable = { var = OCfA_infra_divisor value = 1.0 }

		IF = {
			LIMIT = { check_variable = { var = OCfA_infra_strain value = 0 compare = less_than } }
			set_variable = { var = OCfA_infra_strain value = 0 }
		}
		add_to_temp_variable = { var = OCfA_infra_divisor value = OCfA_infra_strain }
		divide_variable = { var = OCfA_calculated_battalion_capacity value = OCfA_infra_divisor }
	}
	ELSE = {
		set_variable = { var = OCfA_infra_strain value = 0 }
	}

	# --- STEP 4: FINAL ADJUSTMENTS & CLAMPING ---
	add_to_variable = { var = OCfA_calculated_battalion_capacity value = modifier@OCfA_battalion_limit_bonus }

	IF = {
		LIMIT = { is_major = yes }
		multiply_variable = { var = OCfA_calculated_battalion_capacity value = @OCfA_is_major_mult }
	}
	IF = {
		LIMIT = { has_war = yes }
		multiply_variable = { var = OCfA_calculated_battalion_capacity value = @OCfA_has_war_mult }
	}

	clamp_variable = { var = OCfA_calculated_battalion_capacity min = @OCfA_min_capacity max = @OCfA_max_capacity }
	round_variable = OCfA_calculated_battalion_capacity

	# --- STEP 5: CALCULATE BATTALION LOAD ---
	IF = {
		LIMIT = { has_game_rule = { rule = OCfA_weight_rule option = WEIGHT_ON } }
		set_temp_variable = { var = OCfA_current_battalion_load value = 0 }

		set_variable = { var = OCfA_bat_inf value = num_battalions_with_type@infantry }
		multiply_variable = { var = OCfA_bat_inf value = @OCfA_weight_infantry }
		add_to_temp_variable = { var = OCfA_current_battalion_load value = OCfA_bat_inf }

		set_variable = { var = OCfA_bat_lar value = num_battalions_with_type@light_armor }
		multiply_variable = { var = OCfA_bat_lar value = @OCfA_weight_light_armor }
		add_to_temp_variable = { var = OCfA_current_battalion_load value = OCfA_bat_lar }

		set_variable = { var = OCfA_bat_mar value = num_battalions_with_type@medium_armor }
		multiply_variable = { var = OCfA_bat_mar value = @OCfA_weight_medium_armor }
		add_to_temp_variable = { var = OCfA_current_battalion_load value = OCfA_bat_mar }

		set_variable = { var = OCfA_bat_har value = num_battalions_with_type@heavy_armor }
		multiply_variable = { var = OCfA_bat_har value = @OCfA_weight_heavy_armor }
		add_to_temp_variable = { var = OCfA_current_battalion_load value = OCfA_bat_har }

		set_variable = { var = OCfA_bat_mot value = num_battalions_with_type@motorized }
		multiply_variable = { var = OCfA_bat_mot value = @OCfA_weight_motorized }
		add_to_temp_variable = { var = OCfA_current_battalion_load value = OCfA_bat_mot }

		set_variable = { var = OCfA_bat_mec value = num_battalions_with_type@mechanized }
		multiply_variable = { var = OCfA_bat_mec value = @OCfA_weight_mechanized }
		add_to_temp_variable = { var = OCfA_current_battalion_load value = OCfA_bat_mec }

		# --- MISCELLANEOUS SHIPS ---
		set_temp_variable = { var = OCfA_total_special_battalion value = num_battalions_with_type@infantry }
		add_to_temp_variable = { var = OCfA_total_special_battalion value = num_battalions_with_type@light_armor }
		add_to_temp_variable = { var = OCfA_total_special_battalion value = num_battalions_with_type@medium_armor }
		add_to_temp_variable = { var = OCfA_total_special_battalion value = num_battalions_with_type@heavy_armor }
		add_to_temp_variable = { var = OCfA_total_special_battalion value = num_battalions_with_type@motorized }
		add_to_temp_variable = { var = OCfA_total_special_battalion value = num_battalions_with_type@mechanized }

		set_variable = { var = OCfA_bat_misc value = num_battalions }
		subtract_from_variable = { var = OCfA_bat_misc value = OCfA_total_special_battalion }
		add_to_temp_variable = { var = OCfA_current_battalion_load value = OCfA_bat_misc }

		set_variable = { var = OCfA_current_battalion_load value = OCfA_current_battalion_load }
	}
	ELSE_IF = {
		multiply_variable = { var = OCfA_calculated_battalion_capacity value = 1.5 }
		round_variable = OCfA_calculated_battalion_capacity

		set_variable = { var = OCfA_current_battalion_load value = num_battalions }
	}

	set_variable = { var = OCfA_calculated_demob_battalion_capacity value = OCfA_calculated_battalion_capacity }
	multiply_variable = { var = OCfA_calculated_demob_battalion_capacity value = 1.02 }
	add_to_temp_variable = { var = OCfA_calculated_demob_battalion_capacity value = 25 }
	round_variable = OCfA_calculated_demob_battalion_capacity

	IF = { # --- Demob ---
		LIMIT = {
			has_war = no
			check_variable = { OCfA_current_battalion_load > OCfA_calculated_demob_battalion_capacity }
			NOT = {
				has_active_mission = demobilization_military_mission
				has_country_flag = OCfA_recent_demobilization
				has_global_flag = OCfA_prevent_demobilization
			}
		}
		activate_mission = demobilization_military_mission
	}

	# --- STEP 6: CURVE ---
	set_temp_variable = { var = OCfA_capacity_diff value = OCfA_calculated_battalion_capacity }
	subtract_from_temp_variable = { var = OCfA_capacity_diff value = OCfA_current_battalion_load }
	divide_temp_variable = { var = OCfA_capacity_diff value = OCfA_calculated_battalion_capacity }
	multiply_temp_variable = { var = OCfA_capacity_diff value = 100 }

	OCfA_calculate_battalion_capacity_linear = yes
	IF = {
		LIMIT = { has_game_rule = { rule = OCfA_buff_rule option = BUFF_ON } }
		clamp_temp_variable = { var = OCfA_curve_value min = -1.0 max = 1.0 }
	}
	ELSE = {
		clamp_temp_variable = { var = OCfA_curve_value min = -1.0 max = 0.0 }
	}
	multiply_temp_variable = { var = OCfA_curve_value value = 0.8 }
	set_temp_variable = { var = OCfA_curve_value_inversion value = OCfA_curve_value }
	multiply_temp_variable = { var = OCfA_curve_value_inversion value = -1 }

	set_variable = { var = OCfA_ttf_ocd_value_inversion value = OCfA_curve_value_inversion }
	multiply_variable = { var = OCfA_ttf_ocd_value_inversion value = 1.0 }

	set_variable = { var = OCfA_moz_ocd_value value = OCfA_curve_value }
	multiply_variable = { var = OCfA_moz_ocd_value value = 0.5 }

	set_variable = { var = OCfA_amf_ocd_value value = OCfA_curve_value }
	multiply_variable = { var = OCfA_amf_ocd_value value = 0.5 }

	set_variable = { var = OCfA_pls_ocd_value value = OCfA_curve_value }
	multiply_variable = { var = OCfA_pls_ocd_value value = 0.5 }

	# --- STEP 7: UI PROMOTION & DEBUGGING ---
	set_variable = { var = OCfA_pop_core_weight_value value = @OCfA_pop_core_weight }
	set_variable = { var = OCfA_pop_non_core_weight_value value = @OCfA_pop_non_core_weight }
	set_variable = { var = OCfA_pop_divisor_value value = @OCfA_pop_divisor }
	set_variable = { var = OCfA_state_weight_value value = @OCfA_state_weight }
	set_variable = { var = OCfA_mil_weight_value value = @OCfA_mil_weight }
	set_variable = { var = OCfA_civ_weight_value value = @OCfA_civ_weight }
	set_variable = { var = OCfA_doc_weight_value value = @OCfA_dockyard_weight }

	set_variable = { var = OCfA_infantry_weight_value value = @OCfA_weight_infantry }
	set_variable = { var = OCfA_light_armor_weight_value value = @OCfA_weight_light_armor }
	set_variable = { var = OCfA_medium_armor_weight_value value = @OCfA_weight_medium_armor }
	set_variable = { var = OCfA_heavy_armor_weight_value value = @OCfA_weight_heavy_armor }
	set_variable = { var = OCfA_motorized_weight_value value = @OCfA_weight_motorized }
	set_variable = { var = OCfA_mechanized_weight_value value = @OCfA_weight_mechanized }

	set_variable = { var = OCfA_industry_scaling_multiplier_val value = @industry_scaling_multiplier }
	set_variable = { var = OCfA_min_industry_factor_val value = @min_industry_factor }
	set_variable = { var = OCfA_is_major_mult_value value = @OCfA_is_major_mult }
	set_variable = { var = OCfA_has_war_mult_value value = @OCfA_has_war_mult }

	set_variable = { var = OCfA_min_capacity_value value = @OCfA_min_capacity }
	set_variable = { var = OCfA_max_capacity_value value = @OCfA_max_capacity }

	IF = {
		LIMIT = { is_debug = yes }
		OCfA_calculate_battalion_capacity_debug = yes
	}
}
