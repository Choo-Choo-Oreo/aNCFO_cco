update_religious_stance = { # +Map Update
	log = "[GetDateText]: [Root.GetName]: update_religious_stance"
	if = {
		limit = { has_cosmetic_tag = VLADAROS_IMPERIOUS_UNIFIED }
		add_ideas = RoF_vladaros_cult_religion_law
	}
	# Special Countries Countries First
	else_if = {
		limit = { has_government = unitism }
		add_ideas = RoF_ideological_religion_law
	}
	else_if = {
		limit = { has_country_flag = harmonainus_religion }
		add_ideas = RoF_harmonainus_religion_law
	}
	else_if = {
		limit = { has_country_flag = harmonainus_reformed_religion }
		add_ideas = RoF_harmonainus_reformed_religion_law
	}
	else_if = {
		limit = { has_country_flag = pullus_religion }
		add_ideas = RoF_pullus_religion_law
	}
	else_if = {
		limit = { has_country_flag = ideological_religion }
		add_ideas = RoF_ideological_religion_law
	}
	else_if = {
		limit = { has_country_flag = emperor_personality_religion }
		add_ideas = RoF_emperor_personality_religion_law
	}
	else_if = {
		limit = { has_country_flag = thalassaus_religion }
		add_ideas = RoF_thalassaus_religion_law
	}
	else_if = {
		limit = { has_country_flag = ager_montis_religion }
		add_ideas = RoF_ager_montis_religion_law
	}
	else_if = {
		limit = { has_country_flag = animaekin_religion }
		add_ideas = RoF_animaekin_religion_law
	}
	else_if = { # Currently is planned with no name.
		limit = { has_country_flag = yellow_reg_religion }
		add_ideas = RoF_yellow_reg_religion_law
	}
	else_if = {
		limit = { has_country_flag = secular_religion }
		add_ideas = RoF_secular_religion_law
	}
	else_if = { # Just to have a default.
		limit = {
			OR = {
				has_country_flag = nihilism_religion
				NOT = {
					OR = {
						has_country_flag = harmonainus_religion
						has_country_flag = harmonainus_reformed_religion
						has_country_flag = pullus_religion
						has_country_flag = emperor_personality_religion
						has_country_flag = thalassaus_religion
						has_country_flag = ager_montis_religion
						has_country_flag = animaekin_religion
						has_country_flag = yellow_reg_religion
						has_country_flag = secular_religion
					}
				}
			}
		}
		add_ideas = RoF_nihilism_religion_law
	}
	force_update_map_mode = { # We update with it here to reduce lag.
		limit = { is_ai = no }
		mapmode = country_religions_map
	}
}
