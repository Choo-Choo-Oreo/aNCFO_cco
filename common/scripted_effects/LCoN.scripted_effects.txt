# --- GLOBAL WEIGHTS ---
# Determines the logistics contribution provided by different types of industry.
@dockyard_weight = 20
@mil_weight = 1.0
@civ_weight = 1.0

# --- QUADRATIC & SIGMOID SETTINGS ---
# Controls the steepness of the curve; a higher value makes the penalty/bonus ramp up more slowly.
@scaling_divisor = 5000
# Sets the maximum percentage of bonus or penalty a country can receive per dockyard owned (Specific to the Sigmoid formula).
@max_modifier_per_dockyard = 0.2

# --- LINEAR SETTINGS ---
# Controls the steepness of the linear scaling; higher values result in a gentler slope (1.0 / Divisor).
@linear_divisor = 100

# --- LOGARITHMIC SETTINGS ---
# The maximum possible penalty/bonus (asymptote). 
@log_max_limit = 5.0 
# Higher values make the curve flatten faster (reach the limit sooner).
@log_flattening = 100

# --- STEP SETTINGS ---
# The three thresholds for the "Break-it" steps (Distance from Cap).
@step_threshold_1 = -1
@step_threshold_2 = -50
@step_threshold_3 = -100
# The fixed penalties for each threshold.
@step_penalty_1 = -0.25
@step_penalty_2 = -0.50
@step_penalty_3 = -1.0

# --- HYPERBOLIC SETTINGS ---
# The "Critical Mass" buffer. If you go this many ships over cap, penalty becomes extreme/infinite.
@hyper_buffer = 100 
# Scaling factor to keep numbers readable within the engine.
@hyper_scaler = 1

calculate_naval_capacity_logistics_linear = {
	# --- STEP 1: LINEAR SCALING ---
	set_temp_variable = { var = curve_value value = capacity_diff }
	divide_temp_variable = { var = curve_value value = @linear_divisor } 

	# --- STEP 2: RESTORE DIRECTION ---
	# Note: Unlike the quadratic/cubic calculation, linear naturally preserves 
	# the sign. We include this check to ensure parity with the other formulas.
	if = {
		limit = { 
			check_variable = { var = capacity_diff value = 0 compare = less_than } 
			# Ensure the value is negative if we are over capacity
			check_variable = { var = curve_value value = 0 compare = greater_than } 
		}
		multiply_temp_variable = { var = curve_value value = -1 }
	}
}

calculate_naval_capacity_logistics_quadratic = {
	# --- STEP 1: CREATE THE CURVE ---
	set_temp_variable = { var = curve_value value = capacity_diff }
	multiply_temp_variable = { var = curve_value value = capacity_diff }
	
	# --- STEP 2: SCALING ---
	divide_temp_variable = { var = curve_value value = @scaling_divisor }

	# --- STEP 3: RESTORE DIRECTION ---
	if = {
		limit = { check_variable = { var = capacity_diff value = 0 compare = less_than } }
		multiply_temp_variable = { var = curve_value value = -1 }
	}
}

calculate_naval_capacity_logistics_sigmoid = {
	calculate_naval_capacity_logistics_quadratic = yes

	# --- STEP 4: SIGMOID CLAMP ---
	set_temp_variable = { var = logic_cap value = num_of_naval_factories }
	multiply_temp_variable = { var = logic_cap value = @max_modifier_per_dockyard }
	
	if = {
		limit = { check_variable = { var = curve_value value = logic_cap compare = greater_than } }
		set_temp_variable = { var = curve_value value = logic_cap }
	}

	set_temp_variable = { var = logic_floor value = logic_cap }
	multiply_temp_variable = { var = logic_floor value = -1 }
	if = {
		limit = { check_variable = { var = curve_value value = logic_floor compare = less_than } }
		set_temp_variable = { var = curve_value value = logic_floor }
	}
}

calculate_naval_capacity_logistics_logarithmic = {
	# --- DIMINISHING RETURNS CURVE (Michaelis-Menten approximation) ---
	# Math: Modifier = (Max_Limit * Diff) / (Diff + Flattening_Factor)
	# Result: Grows fast at first, then flattens out forever.
	# Preserve sign for direction calculation later
	set_temp_variable = { var = is_negative value = 0 }
	if = {
		limit = { check_variable = { var = capacity_diff value = 0 compare = less_than } }
		multiply_temp_variable = { var = capacity_diff value = -1 }
		set_temp_variable = { var = is_negative value = 1 }
	}

	set_temp_variable = { var = denominator value = capacity_diff }
	add_to_temp_variable = { var = denominator value = @log_flattening }
	
	set_temp_variable = { var = curve_value value = @log_max_limit }
	multiply_temp_variable = { var = curve_value value = capacity_diff }
	divide_temp_variable = { var = curve_value value = denominator }

	# Restore Direction (Negative means Over Cap = Penalty)
	if = {
		limit = { check_variable = { var = is_negative value = 1 } }
		multiply_temp_variable = { var = curve_value value = -1 }
	}
}

calculate_naval_capacity_logistics_step = {
	# Default: Small bonus if under cap
	set_temp_variable = { var = curve_value value = 0.1 }

	# Step 1 Check
	if = {
		limit = { check_variable = { var = capacity_diff value = @step_threshold_1 compare = less_than_or_equals } }
		set_temp_variable = { var = curve_value value = @step_penalty_1 }
	}
	# Step 2 Check
	if = {
		limit = { check_variable = { var = capacity_diff value = @step_threshold_2 compare = less_than_or_equals } }
		set_temp_variable = { var = curve_value value = @step_penalty_2 }
	}
	# Step 3 Check
	if = {
		limit = { check_variable = { var = capacity_diff value = @step_threshold_3 compare = less_than_or_equals } }
		set_temp_variable = { var = curve_value value = @step_penalty_3 }
	}
}

calculate_naval_capacity_logistics_hyperbolic = {
	# If Under Cap: Standard Linear Bonus
	if = {
		limit = { check_variable = { var = capacity_diff value = 0 compare = greater_than_or_equals } }
		set_temp_variable = { var = curve_value value = capacity_diff }
		divide_temp_variable = { var = curve_value value = @linear_divisor }
	}
	# If Over Cap: The Death Spiral
	else = {
		set_temp_variable = { var = over_amount value = capacity_diff }
		multiply_temp_variable = { var = over_amount value = -1 } # Keep capacity_diff original, use over_amount
		
		set_temp_variable = { var = den value = @hyper_buffer }
		subtract_from_temp_variable = { var = den value = over_amount }
		
		# Safety: If Denominator is 0 or less (Hit the wall), cap the penalty
		if = {
			limit = { check_variable = { var = den value = 0.1 compare = less_than } }
			set_temp_variable = { var = curve_value value = -10.0 } # Hard 1000% penalty
		}
		else = {
			set_temp_variable = { var = curve_value value = @hyper_scaler }
			divide_temp_variable = { var = curve_value value = den }
			multiply_temp_variable = { var = curve_value value = -1 }
		}
	}
}

calculate_naval_capacity_logistics_debug = {
	# Promote weight constants so they can be viewed in tooltips
	set_variable = { var = num_battleship		value = num_ships_with_type@battleship }
	set_variable = { var = num_carrier			value = num_ships_with_type@carrier }
	set_variable = { var = num_destroyer		value = num_ships_with_type@destroyer }
	set_variable = { var = num_heavy_cruiser	value = num_ships_with_type@heavy_cruiser }
	set_variable = { var = num_light_cruiser	value = num_ships_with_type@light_cruiser }
	set_variable = { var = num_submarine		value = num_ships_with_type@submarine }

	log = "[Root.GetTag] Naval Cap: [?calculated_naval_cap] | Ships: [?num_ships] | Mod: [?LCoN_icd_ncd_value]"
	log = "    Current Consumer Goods: [?current_cg_total|%Y] | Logistical Efficiency: [?econ_efficiency_factor|%G]"
	log = "    Battleship: [?num_battleship] | Carriers: [?num_carrier] | Destroyers: [?num_destroyer] | Heavy Cruisers: [?num_heavy_cruiser] | Light Cruisers: [?num_light_cruiser] | Submarines: [?num_submarine]"
}

calculate_naval_capacity_logistics_formula = {
	# --- STEP 1: INITIAL RAW CAPACITY ---
	set_temp_variable = { var = calculated_naval_cap value = num_of_naval_factories }
	multiply_temp_variable = { var = calculated_naval_cap value = @dockyard_weight }
	
	set_temp_variable = { var = temp_mil_factories value = num_of_military_factories }
	multiply_temp_variable = { var = temp_mil_factories value = @mil_weight }

	set_temp_variable = { var = temp_civ_factories value = num_of_civilian_factories }
	multiply_temp_variable = { var = temp_civ_factories value = @civ_weight }
	
	add_to_temp_variable = { var = calculated_naval_cap value = temp_mil_factories }
	add_to_temp_variable = { var = calculated_naval_cap value = temp_civ_factories }

	round_temp_variable = calculated_naval_cap

	# --- STEP 2: ECONOMIC MOBILIZATION SCALING ---
	# Calculate efficiency first so it can affect BOTH Cap and Curve
	set_temp_variable = { var = econ_efficiency_factor value = 1.0 }

	IF = { # --- ECONOMIC MOBILIZATION SCALING ---
		LIMIT = { has_game_rule = { rule = LCoN_mobilization_scaling_rule option = MOB_ON } }
		
		# 1. Check standard factor (Usually Stability)
		set_temp_variable = { var = check_factor value = modifier@consumer_goods_factor }
		log = "[Root.GetTag] DIAGNOSTIC: Raw Factor = [?check_factor]"

		# 2. Check the "Factories" version (Often where Spirits live)
		set_temp_variable = { var = check_factories_modifier value = modifier@consumer_goods_factories }
		log = "[Root.GetTag] DIAGNOSTIC: Factories Modifier = [?check_factories_modifier]"

		# 1. Base CG from Law
		set_temp_variable = { var = current_cg_total value = modifier@consumer_goods_expected_value }
		log = "[Root.GetTag] LOG 1: Base Law Value = [?current_cg_total]"
		
		# Create the multiplier: (1 + consumer_goods_factor)
		set_temp_variable = { var = cg_multiplier value = 1.0 }
		add_to_temp_variable = { var = cg_multiplier value = modifier@consumer_goods_factor }
		log = "[Root.GetTag] LOG 2: Multiplier (1 + Factor) = [?cg_multiplier]"
		
		# Final Multiply: Base * Multiplier
		multiply_temp_variable = { var = current_cg_total value = cg_multiplier }
		log = "[Root.GetTag] LOG 3: Resulting CG Total = [?current_cg_total]"

		# Apply to Efficiency
		subtract_from_temp_variable = { var = econ_efficiency_factor value = current_cg_total }
		clamp_temp_variable = { var = econ_efficiency_factor min = 0.0 max = 1.0 }
		log = "[Root.GetTag] LOG 4: Final Efficiency Factor = [?econ_efficiency_factor]"

		# Throttling the Cap
		multiply_temp_variable = { var = calculated_naval_cap value = econ_efficiency_factor }
		log = "[Root.GetTag] LOG 5: Throttling the Cap [?calculated_naval_cap]"
		
		# Promote for UI
		set_variable = { var = current_cg_total value = current_cg_total }
		set_variable = { var = econ_efficiency_factor value = econ_efficiency_factor }
		log = "[Root.GetTag] LOG 6: Promote for UI (current_cg_total) [?current_cg_total]"
		log = "[Root.GetTag] LOG 7: Promote for UI (econ_efficiency_factor) [?econ_efficiency_factor]"
	}

	# --- STEP 3: RUN THE CHOSEN CURVE ---
	set_temp_variable = { var = capacity_diff value = calculated_naval_cap }
	subtract_from_temp_variable = { var = capacity_diff value = num_ships }
	IF = {
		LIMIT = { has_game_rule = { rule = LCoN_formula_selection option = LINEAR } }
		calculate_naval_capacity_logistics_linear = yes
	}
	ELSE_IF = { 
		LIMIT = { has_game_rule = { rule = LCoN_formula_selection option = QUADRATIC } }
		calculate_naval_capacity_logistics_quadratic = yes
	}
	ELSE_IF = {
		LIMIT = { has_game_rule = { rule = LCoN_formula_selection option = SIGMOID } }
		calculate_naval_capacity_logistics_sigmoid = yes
	}
	ELSE_IF = {
		LIMIT = { has_game_rule = { rule = LCoN_formula_selection option = LOGARITHMIC } }
		calculate_naval_capacity_logistics_logarithmic = yes
	}
	ELSE_IF = {
		LIMIT = { has_game_rule = { rule = LCoN_formula_selection option = STEP } }
		calculate_naval_capacity_logistics_step = yes
	}
	ELSE_IF = {
		LIMIT = { has_game_rule = { rule = LCoN_formula_selection option = HYPERBOLIC } }
		calculate_naval_capacity_logistics_hyperbolic = yes
	}

	# --- STEP 4: FINAL MODIFIER PROCESSING ---
	set_variable = { var = LCoN_icd_ncd_value value = curve_value }

	# Safety Clamp: Prevents dockyards from reaching 0% or negative output
	IF = {
		LIMIT = { has_game_rule = { rule = LCoN_safety_clamp_rule option = CLAMP_ON } }
		clamp_variable = { var = LCoN_icd_ncd_value min = -1.0 }
	}

	# --- STEP 5: VARIABLE PROMOTION & LOGGING ---
	# Promote the throttled cap to a permanent variable for the UI
	set_variable = { var = calculated_naval_cap value = calculated_naval_cap }
	round_variable = calculated_naval_cap

	# Breakdown for the detailed industrial tooltip
	set_variable = { var = dockyard_weighted_value value = num_of_naval_factories }
	multiply_variable = { var = dockyard_weighted_value value = @dockyard_weight }
	
	set_variable = { var = temp_mil_factories value = num_of_military_factories }
	multiply_variable = { var = temp_mil_factories value = @mil_weight }
	
	set_variable = { var = temp_civ_factories value = num_of_civilian_factories }
	multiply_variable = { var = temp_civ_factories value = @civ_weight }

	# Promote weight constants so they can be viewed in tooltips
	set_variable = { var = doc_weight_value value = @dockyard_weight }
	set_variable = { var = mil_weight_value value = @mil_weight }
	set_variable = { var = civ_weight_value value = @civ_weight }

	IF = {
		LIMIT = { is_debug = yes }
		calculate_naval_capacity_logistics_debug = yes
	}
}
