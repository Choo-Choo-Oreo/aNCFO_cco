HHE_has_war_with_HLF = {
	IF = { # HHE has war with HLF
		LIMIT = {
			TAG = HHE
			HHE = { has_war_with = HLF }
			HLF = { has_capitulated = no }
		}
		set_country_flag = ignore_micro_scripts
		ELSE = { 
			LIMIT = { TAG = HHE }
			clr_country_flag = ignore_micro_scripts
		}
	}
}
