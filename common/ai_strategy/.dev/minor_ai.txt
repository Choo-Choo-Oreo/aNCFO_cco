			has_manpower < 1000000

				original_tag = RAJ

				has_equipment = { infantry_equipment < 50000 }
			}
		}
		OR = {
			has_war = no
			AND = {
				has_war = yes
				enemies_strength_ratio > 1.2
			}
		}
	}

		OR = {
			is_historical_focus_on = no

				has_manpower > 1500000
				has_army_manpower = { size > 1500000 }

					AND = {
						original_tag = RAJ

							has_manpower < 2000000
							has_equipment = { infantry_equipment < 50000 }
						}
					}
				}
			}
			enemies_strength_ratio < 1.0
		}
	}


	ai_strategy = {
		type = force_concentration_factor
		value = -100
	}

	ai_strategy = {

		ratio = 0.3
