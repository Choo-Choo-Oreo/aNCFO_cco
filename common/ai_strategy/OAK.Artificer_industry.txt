OAK_industry_industrial_might = {
	allowed = { original_tag = OAK }
	enable = {
		set_temp_variable = { total_industry = building_level@arms_factory }
		add_to_temp_variable = { total_industry = building_level@dockyard }
		check_variable = { var = total_industry value = industrial_complex_level compare = greater_than }
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = building_target
		id = industrial_complex
		value = 100
	}
}

OAK_industry_target_500_industrial_complex = {
	allowed = {
		OR = {
			original_tag = COA
			original_tag = OAK
		}
	}
	enable = {
		is_late_game = yes
		NOT = { check_variable = { num_of_civilian_factories = 500 } }
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = building_target
		id = industrial_complex
		value = 500
	}
}
