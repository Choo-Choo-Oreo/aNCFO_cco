MJU_defend_UFS_against_threats = {
	allowed = { original_tag = MJU original_tag = UFS }
	enable = {
		HHE = { has_war_with = UFS }
		97 = { is_controlled_by = UFS }
		country_exists = UFS
	}
	abort = {
		OR = {
			NOT = {
				country_exists = UFS 
				97 = { is_controlled_by = UFS }
			}
			has_war_with = UFS
		}
	}
	
	ai_strategy = {
		type = befriend
		id = "UFS"
		value = 200
	}
	ai_strategy = {
		type = alliance
		id = "UFS"
		value = 200
	}
	ai_strategy = {
		type = protect
		id = "UFS"
		value = 200
	}
}
