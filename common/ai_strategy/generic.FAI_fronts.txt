# Homeland Defense Logic
FAI_fronts_defend_your_home_please = {
    enable = {
        surrender_progress > 0.2
    }
    abort_when_not_enabled = yes

    ai_strategy = {
        type = spare_unit_factor
        value = -50
    }

    ai_strategy = {
        type = garrison
        value = -100
    }
}

FAI_fronts_homeland_defense = {
    enable = {
        NOT = {
            OR = {
                has_country_flag = ignore_micro_scripts 
                has_global_flag = ignore_micro_scripts_global
            }
        }
        has_war = yes
        surrender_progress > 0.35
    }

    abort_when_not_enabled = yes
    
    ai_strategy = {
        type = front_unit_request
        state_trigger = {
            any_state = {
                is_core_of = ROOT
                owner = {
                    has_war_with = ROOT
                }
            }
        }
        value = 120                         # Will be added as a factor over regular requests.
    }
}
# Frontline Management
FAI_fronts_management_rush = {
    enable = {
        NOT = {
            OR = {
                has_country_flag = ignore_micro_scripts 
                has_global_flag = ignore_micro_scripts_global
                surrender_progress > 0.35
                has_manpower < 1000
            }
        }
        has_war = yes
        is_major = no
    }

    abort_when_not_enabled = yes
    
    ai_strategy = {
        type = front_control

        country_trigger = { 
            enemies_strength_ratio < 1.6
        }

        priority = 100
        ordertype = front
        execution_type = rush
        execute_order = yes
        manual_attack = no
    }
}
