FAI_army_use_surplus_equipment_for_infantry = {
	enable = {
		#has_reached_maximum_divisions = no
		check_variable = { var = reserved value = deployed_total compare = greater_than }
		check_variable = { num_equipment@anti_air_equipment > num_equipment_in_armies@anti_air_equipment }
		check_variable = { num_equipment@anti_tank_equipment > num_equipment_in_armies@anti_tank_equipment }
		check_variable = { num_equipment@infantry_equipment > num_equipment_in_armies@infantry_equipment }

		check_variable = { num_equipment@support_equipment > num_equipment_in_armies@support_equipment }

		type = role_ratio
		id = infantry
		value = 200
	}

	}
}
