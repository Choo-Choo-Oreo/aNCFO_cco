division_early_war_to_late_peace = {
	allowed = {
		OR = {
			TAG = MTR TAG = ROM
			TAG = TRH TAG = UEK
		}
	}
	enable = {
		OR = {
			AND = {
				OR = { TAG = MTR TAG = ROM }
				NOT = { AND = { has_full_control_of_state = 434 has_full_control_of_state = 629 } }
			}
			AND = {
				OR = { TAG = TRH TAG = UEK }
				NOT = { AND = { has_full_control_of_state = 790 has_full_control_of_state = 803 } }
			}
		}
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = ai_wanted_divisions_factor
		value = 550
	}
}
