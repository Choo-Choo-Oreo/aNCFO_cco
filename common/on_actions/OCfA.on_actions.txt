on_actions = {
	on_startup = {
		effect = { # Scripted Triggers Initialization
			set_global_flag = is_OCfA_enabled
			
			IF = {
				LIMIT = {
					has_game_rule = { rule = OCfA_mobilization_scaling_rule option = MOB_ON }
				}
				set_global_flag = is_OCfA_mobilization_scaling_rule_enabled
			}
		}

		effect = {
			every_possible_country = {
				IF = {
					LIMIT = {
						NOT = { has_dynamic_modifier = { modifier = OCfA_opereation_capacity_dynamic_modifier } }
					}
					add_dynamic_modifier = { modifier = OCfA_opereation_capacity_dynamic_modifier }
				}
				OCfA_calculate_battalion_capacity_formula = yes
			}
			set_global_flag = { flag = OCfA_prevent_demobilization days = 730 value = 1 }
		}
	}

	on_weekly = { # on_monthly is to slow for the rapid change of unit's
		effect = {
			IF = {
				LIMIT = {
					NOT = { has_dynamic_modifier = { modifier = OCfA_opereation_capacity_dynamic_modifier } }
				}
				add_dynamic_modifier = { modifier = OCfA_opereation_capacity_dynamic_modifier }
			}
			OCfA_calculate_battalion_capacity_formula = yes
		}
	}

	on_peace = {
		effect = {
			log = "[GetDateText]: [Root.GetName]: on_peace"
			THIS = {
				OCfA_calculate_battalion_capacity_formula = yes
				set_country_flag = OCfA_allow_demobilization
			}
		}
	}
}
