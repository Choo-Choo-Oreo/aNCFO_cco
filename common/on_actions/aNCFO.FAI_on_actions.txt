	on_monthly = { # Triggers each month for every country separately
		effect = { # Building Queuer
			IF = {
				LIMIT = {
					is_ai = yes
					is_erly_game = no
				}
				every_owned_state = { # Oil-Fired Generator Queuing
					IF = {
						LIMIT = {
							check_variable = { var = num_of_factories value = 10 compare = greater_than_or_equals }
						}
						IF = {
							LIMIT = {
							    check_variable = { var = infrastructure_level value = 3 compare = less_than }
							    NOT = { has_state_flag = { flag = bq_oil_fired_generator days < 1 } }
							}
							add_building_construction = { type = infrastructure level = 1 instant_build = no }
							set_state_flag = { flag = bq_oil_fired_generator days = 365 }
							ELSE = {
							    LIMIT = {
							        check_variable = { var = infrastructure_level value = 3 compare = greater_than_or_equals }
							        NOT = { has_state_flag = { flag = bq_oil_fired_generator days < 1 } }
							    }
							    add_building_construction = { type = oil_fired_generator level = 1 instant_build = no }
								set_state_flag = { flag = bq_oil_fired_generator days = 365 }
							}
						}
					}
				}
				IF = { # Construction Sector Queuing
					LIMIT = { num_of_civilian_factories_available_for_projects > 49 }
					every_owned_state = {
						random_select_amount = 3 # "random_select_amount" prevents "construction_complex" Queue overload
						LIMIT = {
							free_building_slots = { building = industrial_complex size < 1 }
							NOT = { has_state_flag = { flag = bq_construction_complex days < 1 } }
						}
						add_building_construction = { type = construction_complex level = 1 instant_build = no }
						set_state_flag = { flag = bq_construction_complex days = 365 }
					}
					ELSE = {
						every_owned_state = {
							random_select_amount = 1 # "random_select_amount" prevents "construction_complex" Queue overload
							LIMIT = {
								free_building_slots = { building = industrial_complex size < 1 }
								NOT = { has_state_flag = { flag = bq_construction_complex days < 1 } }
							}
							add_building_construction = { type = construction_complex level = 1 instant_build = no }
							set_state_flag = { flag = bq_construction_complex days = 365 }
						}
					}
				}
			}
		}
	}
