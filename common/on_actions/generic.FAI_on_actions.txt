# General
	on_startup = { # Trigger the following commands at the first day of a new game, after country selection. Doesn't work with save loading.
		effect = { # Scripted Triggers Initialization
			set_global_flag = is_FantasyAI_enabled

			# Initialize the starting "YEAR" at game starts.
			set_variable = { global.start_year_var = global.year }
			set_variable = { global.four_pass_year_var = global.start_year_var }
			add_to_variable = { global.four_pass_year_var = 4 }
			set_variable = { global.seven_pass_year_var = global.start_year_var }
			add_to_variable = { global.seven_pass_year_var = 7 }

			every_country = {
				FAI_update_colonial_empire = yes
				FAI_update_primarily_coastal = yes
				FAI_update_no_connected_neighbors = yes
				FAI_calculate_resource_status = yes
				FAI_deployed_total = yes
				FAI_convoy_targeter = yes
			}
		}

		effect = { # Democratic "aligned_to_help_countries" startup arrays
			every_country = { # Scope A (Outer)
				limit = {
					has_elections = yes
				}
				# For each country with elections, add all other countries with elections to its array
				every_country = { # Scope B (Inner)
					limit = {
						has_elections = yes
						NOT = { tag = PREV }
					}
					# We are currently in Scope B. 
					# We want to add THIS (Scope B) to PREV's (Scope A) array.
					PREV = {
						if = {
							limit = {
								NOT = { is_in_array = { PREV.aligned_to_help_countries = THIS } }
							}
							add_to_array = { aligned_to_help_countries = PREV }
							log = "[GetDateText]: [THIS.GetName]: Added [PREV.GetName] to attaché targets (democratic startup)"
						}
					}
				}
			}
		}
	}

	on_ruling_party_change = { # old_ideology_token is a temporary variable that stores the old ideology as a token. Alongside what triggers on_government_change, also includes being puppeted or changing the ideology via a console command.
		effect = { # Democratic Target Synchronizer
			every_country = {
				limit = {
					always = yes
				}
				# REMOVE LOGIC
				if = {
					limit = { has_elections = no }
					every_country = { # Scope B: The "Target" countries
						limit = {
							has_elections = yes 
							is_in_array = { aligned_to_help_countries = PREV } # Is the 'Subject' in their array?
						}
						log = "[GetDateText]: [ROOT.GetName]: Removed [PREV.GetName] from attaché targets (no longer has elections)"
						# Remove the 'Subject' (PREV) from the 'Target' (THIS) array
						remove_from_array = { aligned_to_help_countries = PREV }
					}
				}
				# ADD LOGIC
				if = {
					limit = { has_elections = yes }
					every_country = { # Scope B: The "Target" countries
						limit = {
							has_elections = yes
							NOT = { tag = PREV } # Don't add yourself to your own array
							NOT = { is_in_array = { aligned_to_help_countries = PREV } }
						}
						log = "[GetDateText]: [ROOT.GetName]: Added [PREV.GetName] from attaché targets (has elections)"
						# Add the 'Subject' (PREV) to the 'Target' (THIS) array
						add_to_array = { aligned_to_help_countries = PREV }
					}
				}
			}
		}
	}

	on_war = { # THIS is country that has just gotten into a war.
		effect = {
			# ROOT = country declaring, FROM = target
			
			# Check if ROOT has allies that might want to send attachés
			ROOT = {
				every_allied_country = {
					limit = {
						#is_ai = yes
						NOT = { has_war_with = FROM }
						NOT = { is_in_array = { aligned_to_help_countries = ROOT } }
					}
					log = "[GetDateText]: [This.GetName]: Added [Root.GetName] to attaché targets (ally at war)"
					add_to_array = { aligned_to_help_countries = ROOT }
				}
			}
			
			# Same for FROM's allies
			FROM = {
				every_allied_country = {
					limit = {
						#is_ai = yes
						NOT = { has_war_with = ROOT }
						NOT = { is_in_array = { aligned_to_help_countries = FROM } }
					}
					log = "[GetDateText]: [This.GetName]: Added [From.GetName] to attaché targets (ally at war)"
					add_to_array = { aligned_to_help_countries = FROM }
				}
			}
		}

		effect = {
			# ROOT = attacker, FROM = defender
			
			# If this is a major war, all majors should consider sending attachés
			if = {
				limit = {
					OR = {
						ROOT = { is_major = yes }
						FROM = { is_major = yes }
					}
				}
				
				every_country = {
					limit = {
						is_ai = yes
						is_major = yes
						NOT = { has_war_with = ROOT }
						NOT = { has_war_with = FROM }
					}
					
					# Add attacker if not at war with them
					if = {
						limit = {
							NOT = { has_war_with = ROOT }
							NOT = { is_in_array = { aligned_to_help_countries = ROOT } }
						}
						add_to_array = { aligned_to_help_countries = ROOT }
					}
					
					# Add defender if not at war with them
					if = {
						limit = {
							NOT = { has_war_with = FROM }
							NOT = { is_in_array = { aligned_to_help_countries = FROM } }
						}
						add_to_array = { aligned_to_help_countries = FROM }
					}
				}
			}
		}
	}
