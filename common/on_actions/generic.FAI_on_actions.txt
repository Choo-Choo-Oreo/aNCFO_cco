on_actions = {
# General
	on_startup = { # Trigger the following commands at the first day of a new game, after country selection. Doesn't work with save loading.
		effect = {
			set_global_flag = is_CrC_yes # Is this mod on, yes...
			set_global_flag = disable_tag_related_strats # For Smarter AI by TheGamingNot
            # Initialize the counter to 0 when game starts
            set_variable = { global.day_counter = 0 }
		}

		effect = { # Army Size Checker
			# Compute on_monthly_deployed_total = army + navy + airforce
			set_temp_variable = { on_monthly_deployed_total = deployed_army_manpower_k }
			add_to_temp_variable = { on_monthly_deployed_total = deployed_navy_manpower_k }
			add_to_temp_variable = { on_monthly_deployed_total = deployed_airforce_manpower_k }
			# Compute on_monthly_reserved = manpower_k - on_monthly_deployed_total
			set_temp_variable = { on_monthly_reserved = manpower_k }
			subtract_from_temp_variable = { on_monthly_reserved = on_monthly_deployed_total }
			set_variable = { var = deployed_total value = on_monthly_deployed_total }
			set_variable = { var = reserved value = on_monthly_reserved }
			# Used for "check_variable = { var = reserved value = deployed_total compare = greater_than }"

	        # Store deployed fighters in temp variable first
	        set_temp_variable = { deployed_fighters = num_deployed_planes_with_type@fighter }
	        set_temp_variable = { stockpile_planes = num_equipment@small_plane_airframe }
		}

		effect = { # Calculate the "Convoy Target" based on total deployed manpower
			set_variable = { var = FAI_convoy_target value = deployed_total }
			#multiply_variable = { var = FAI_convoy_target value = 1 }
			if = { # Always have 1k convoys as major
				limit = {
					ROOT = { is_major = yes }
					check_variable = { var = FAI_convoy_target value = 1000 compare = less_than }
				}
				set_variable = { var = FAI_convoy_target value = 1000 }
			}
			# Compute the Overflow Limit
			set_variable = { var = FAI_convoy_of1_limit value = FAI_convoy_target }
			set_variable = { var = FAI_convoy_of2_limit value = FAI_convoy_target }
			multiply_variable = { var = FAI_convoy_of1_limit value = 1.2 }
			multiply_variable = { var = FAI_convoy_of2_limit value = 2.0 }
		}
	}

	on_daily = { # Triggers each day for every country separately (performance heavy, use carefully)
		effect = {
            # Increment by 1 every day
            add_to_variable = { global.day_counter = 1 }
		}
	}

	on_weekly = { # Triggers each week for every country separately

		effect = { # Building Queuer
			IF = {
				LIMIT = {
					is_ai = yes
					is_erly_game = no
				}
				every_owned_state = {# "Reinforced Electrical Grid"/"High Capacity Electrical Grid" Queuing
					random_select_amount = 1 # "random_select_amount" prevents "energy_infrastructure"/"industrial_infrastructure" Queue overload
					LIMIT = {
						owner = {
							num_of_civilian_factories_available_for_projects > 49
						}
						check_variable = { var = energy_infrastructure value = 1 compare = less_than }
						check_variable = { var = industrial_infrastructure value = 1 compare = less_than }
					}
					IF = { # industrial_infrastructure
						LIMIT = {
							free_building_slots = { building = industrial_infrastructure size > 0 }
							NOT = { has_state_flag = { flag = bq_infrastructure days < 1 } }
							OR = {
								check_variable = { var = resource@oil		value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@coal		value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@aluminium	value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@rubber	value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@tungsten	value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@steel		value = 20 compare = greater_than_or_equals }
								check_variable = { var = resource@chromium	value = 20 compare = greater_than_or_equals }
							}
						}
						add_building_construction = { type = industrial_infrastructure level = 1 instant_build = no }
						set_state_flag = { flag = bq_infrastructure days = 365 }
						ELSE = { # energy_infrastructure
						    LIMIT = {
						    	free_building_slots = { building = energy_infrastructure size > 0 }
						        NOT = { has_state_flag = { flag = bq_infrastructure days < 1 } }
						        OR = {
						        	AND = {
						        		owner = { check_variable = { var = num_of_factories value = 100 compare = greater_than_or_equals } }
						        		check_variable = { var = num_of_factories value = 2 compare = greater_than_or_equals }
						        	}
						        	AND = {
						        		owner = { check_variable = { var = num_of_factories value = 100 compare = less_than_or_equals } }
						        		check_variable = { var = num_of_factories value = 9 compare = greater_than_or_equals }
						        	}
						        }
						    }
						    add_building_construction = { type = energy_infrastructure level = 1 instant_build = no }
							set_state_flag = { flag = bq_infrastructure days = 365 }
						}
					}
				}
				every_owned_state = { # Commercial Nuclear Reactor Queuing
					random_select_amount = 1
					LIMIT = {
						owner = { has_tech = commercial_nuclear_reactor_tech }
						free_building_slots = { building = commercial_nuclear_reactor size > 0 }
						NOT = { has_state_flag = { flag = bq_commercial_nuclear_reactor days < 1 } }
					}
					add_building_construction = { type = commercial_nuclear_reactor level = 1 instant_build = no }
					set_state_flag = { flag = bq_commercial_nuclear_reactor days = 365 }
				}
				every_controlled_state = { # Air Base Queuing
				    random_select_amount = 3
				    LIMIT = {
				    	controller = {
							check_variable = { var = stockpile_planes value = deployed_fighters compare = greater_than }
							check_variable = { var = num_equipment@small_plane_airframe value = 2000 compare = greater_than }
				    	}
				        # Has at least level 1 air base
				        check_variable = { var = building_level@air_base value = 1 compare = greater_than_or_equals }
				        # Not at max level (10)
				        check_variable = { var = building_level@air_base value = 10 compare = less_than }
				        # Not recently queued
				        NOT = { has_state_flag = { flag = bq_air_base days < 365 } }
				        # Has free building slots
				        free_building_slots = {
				        	building = air_base
				        	size > 0
				        }
				    }
				    add_building_construction = {
				        type = air_base
				        level = 1
				        instant_build = no
				    }
				    set_state_flag = {
				        flag = bq_air_base
				        days = 365
				    }
				}
			}
		}
	}
