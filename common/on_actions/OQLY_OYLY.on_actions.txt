on_actions = {
	on_startup = {
		effect = {
			set_variable = { global.quarterly_counter = 3 }
			log = "GLOBAL LOG: Quarterly Pulse Triggered on [GetDateText]. Variable Reset to 3."
			
			set_variable = { global.yearly_counter = 12 }
			log = "GLOBAL LOG: Yearly Pulse Triggered on [GetDateText]. Variable Reset to 12."
		}
	}
	# Change the `ZZZ` to any confirmed scope.
	on_monthly = {
		effect = {
			IF = {
	            LIMIT = { NOT = { has_global_flag = global_monthly_done } }
	            set_global_flag = { flag = global_monthly_done days = 1 value = 1 }

				log = "OQLY / OYLY fire on_monthly_TAG"
				subtract_from_variable = { global.quarterly_counter = 1 }
				subtract_from_variable = { global.yearly_counter = 1 }

				# Place any scripts that relay on ON_QUARTERLY or ON_YEARLY here.

				# --- ON_QUARTERLY (Country Scoped) ---
				IF = {
					LIMIT = { check_variable = { global.quarterly_counter = 0 } }
					
					# --- ON_QUARTERLY ---
					log = "GLOBAL LOG: Quarterly Pulse Triggered on [GetDateText]. Variable Reset to 3."
					
					# --- RESET ---
					set_variable = { global.quarterly_counter = 3 }
				}
				
				# --- ON_YEARLY (Country Scoped) ---
				IF = {
					LIMIT = { check_variable = { global.yearly_counter = 0 } }
					
					# --- ON_YEARLY ---
					log = "GLOBAL LOG: Yearly Pulse Triggered on [GetDateText]. Variable Reset to 12."
					
					# --- RESET ---
					set_variable = { global.yearly_counter = 12 }
				}
			}
		}
	}
}
