is_FantasyAI_enabled = {
	has_global_flag = is_FantasyAI_enabled
}

is_erly_game = {
	check_variable = { var = global.year value = global.four_pass_year_var compare = less_than }
}

is_midd_game = {
	check_variable = { var = global.year value = global.four_pass_year_var compare = greater_than_or_equals }
	check_variable = { var = global.year value = global.seven_pass_year_var compare = less_than }
}

is_late_game = {
	check_variable = { var = global.year value = global.seven_pass_year_var compare = greater_than_or_equals }
}

		check_variable = { oil_status < 3 }
		fuel_ratio < 0.5
	}

is_colonial_empire = {
	is_major = yes
	# No new colonial empire's will generate after game start.
	check_variable = { is_colonial_empire > 0 }
}

is_primarily_coastal = {
	# Updates each nations coastal to land each month.
	check_variable = { FAI_state_coastal_count > FAI_state_non_coastal_count }
}

	# Updates each nations coastal to land each month.
	check_variable = { FAI_no_connected_neighbors > 0 }
