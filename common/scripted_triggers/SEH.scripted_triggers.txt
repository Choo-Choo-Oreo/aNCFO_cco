# args: temp_min_proportion, temp_species_check
SEH_count_proportion_owned_species = {
	hidden_trigger = {
		all_of_scopes = {
			array = global.SEH_species_states_@var:temp_species_check
			tooltip = " "
			IF = {
				limit = {
					is_owned_and_controlled_by = PREV
					is_core_of = PREV
				}
				add_to_temp_variable = { temp_owned_states_count = 1 }
			}
		}
	}
	set_temp_variable = { proportion = temp_owned_states_count }
	divide_temp_variable = { proportion = global.SEH_species_states_@var:temp_species_check^num }

	set_temp_variable = { proportion_display = temp_owned_states_count }
	subtract_from_temp_variable = { proportion_display = 1 }
	divide_temp_variable = { proportion_display = global.SEH_species_states_@var:temp_species_check^num }

	set_temp_variable = { SEH_temp_species = temp_species_check }
	custom_trigger_tooltip = {
		tooltip = SEH_count_proportion_owned_species_tt
		check_variable = { var = proportion value = temp_min_proportion compare = greater_than_or_equals }
	}
}

SEH_has_accepted_species_full = {
	ROOT = {
		set_temp_variable = { var = SEH_temp_species value = SEH_temp_has_species tooltip = SEH_has_accepted_species_full_tt }
		hidden_trigger = {
			is_in_array = { array = SEH_accepted_species_full value = SEH_temp_species }
		}
	}
}
