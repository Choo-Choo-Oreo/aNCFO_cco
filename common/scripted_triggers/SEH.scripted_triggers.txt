# args: temp_min_proportion, temp_species_check
count_proportion_owned_species = {
	set_temp_variable = { temp_owned_states_count = 0 } # for multiple uses of this scripted trigger
	all_of_scopes = {
		array = global.SEH_species_states_@var:temp_species_check
		IF = {
			limit = {
				is_owned_by = PREV
				is_core_of = PREV
			}
			add_to_temp_variable = { temp_owned_states_count = 1 }
		}
	}
	set_temp_variable = { proportion = temp_owned_states_count }
	divide_temp_variable = { proportion = global.SEH_species_states_@var:temp_species_check^num }
	check_variable = { var = proportion value = temp_min_proportion compare = greater_than_or_equals }
}
