shared_focus = { # The Election
	id = HHE_the_election
	icon = GFX_focus_generic_strike_at_democracy2
	prerequisite = { focus = HHE_blame_merchants focus = HHE_blame_junta focus = HHE_blame_the_people }
	relative_position_id = HHE_senate_reforms
	#mutually_exclusive = { focus =  }
	x = 0
	y = 6
	cost = 10
	
	ai_will_do = {
		base = 1
	}

	available_if_capitulated = yes

	completion_reward = {
		add_political_power = -25
		custom_effect_tooltip = ELECTION_THE_LEADER
		if = { # Tradition
			# Gabriel Strongheart wants to continue the norm or harmonain hegemony even though its not as strong as its use to, very little will change and is rather boring, but we hope to create a fun infighting system that is less violent then Archard Vlad's path.
			# Due to it being boring it shell be hard to get this path.
			limit = {
				has_stability > 0.15
				has_country_flag = halt_reforms
				has_country_flag = light_assassination_prevention
			}
			elected_tradition_effect = yes
		}
		else_if = { # Influencer
			# Isabella Delfina wants to procede to a extent of the "Dead Reforms" to impove apon the Imperium, after some improvement she will start reinfluencing regions the imperium once controlled and then some though either politcal civil war or uprising of natives to reinstate imperial rule/influencer.
			# She is more fun than the Tradition path so she is easier to get well at the same time making it hard to end up with the Tradition path.
			limit = {
				has_stability > 0.10
				has_country_flag = finish_reforms
			}
			elected_influencer_effect = yes
		}
		else_if = { # Authority
			# NAME will be mostly internal wars and annexs though politcal means, he works closer with the nobillity of the regions to being people into the fold. Once he is completed with his work he shell fight wars over independed nations to regain Imperial lands.
			# This path is pretty generic so nothing really special.
			limit = {
				has_stability < 0.15
				has_stability > 0.05
				has_country_flag = halt_reforms
				has_country_flag = heavy_assassination_prevention
			}
			elected_authority_effect = yes
		}
		else_if = { # Warmonger
			# Archard Vlad once he gets elected the imperium shell shatter and blow up into civil war, he is the Radical Militarist with views of industrial progress is liberation for the man well capitalism is exploitive of the works such as looking inward at the nobility 
			# This path is histical and can easily happen if the player doesnt select focus correctly.
			elected_warmonger_effect = yes
		}
		hidden_effect = {
			country_event = { id = HHE.1 }
			remove_ideas = HHE_deed_reform_senete
		}
	}
