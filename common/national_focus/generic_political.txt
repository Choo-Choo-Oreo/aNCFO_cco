shared_focus = {
	id = political_semidemocracy_effort
	icon = GFX_
	prerequisite = { focus = political_effort }
	mutually_exclusive = { focus = political_theocratism_effort focus = political_authoritarianism_effort focus = political_unitism_effort }
	available = {
		OR = {
			has_government = pluralism
			has_government = semidemocracy
			pluralism > 0.25
			semidemocracy > 0.25
		}
	}
	x = -4
	y = 1
	relative_position_id = political_effort
	cost = 10

	search_filters = { FOCUS_FILTER_POLITICAL }

	ai_will_do = {
		factor = 100
		modifier = {
			factor = 100
			OR = {
				has_government = pluralism
				has_government = semidemocracy
				pluralism > 0.34
				semidemocracy > 0.34
			}
		}
		modifier = {
			factor = 0.1
			any_neighbor_country = {
					is_major = yes
				OR = { has_government = pluralism has_government = semidemocracy }
			}
			NOT = {
				any_neighbor_country = {
					is_major = yes
					OR = { has_government = theocratism has_government = authoritarianism has_government = unitism }
				}
			}
		}
	}

	completion_reward = {
		add_timed_idea = { idea = modern_state days = 720 }
	}
}
