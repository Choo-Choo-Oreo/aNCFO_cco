SEH_integration_actions = {
	SEH_integration_region = {
		icon = generic_prepare_civil_war
		target_array = occupied_countries
		target_non_existing = yes
		
		cost = 200

		visible = {
			has_core_occupation_modifier = {
				occupied_country_tag = FROM
				modifier = compliance_40
			}
		}

		available = {
			has_core_occupation_modifier = {
				occupied_country_tag = FROM
				modifier = compliance_80
			}
		}

		complete_effect = {
			every_state = {
				limit = {
					is_core_of = FROM
					is_owned_by = ROOT
				}
				ROOT = {
					add_state_core = PREV
				}
			}
		}

		ai_will_do = {
			factor = 0

			modifier = {
				any_state = {
					is_on_same_continent_as = ROOT
					is_core_of = FROM
					is_core_of = ROOT
				}
				add = 50
			}

			modifier = {
				any_state = {
					is_on_same_continent_as = ROOT
					is_core_of = FROM
					is_claimed_by = ROOT
				}
				add = 50
			}
		}
	}

	instantiate_collaboration = {
		icon = generic_prepare_civil_war
		target_array = occupied_countries
		target_non_existing = yes

		visible = {
			has_core_occupation_modifier = {
				occupied_country_tag = FROM
				modifier = compliance_40
			}
		}

		available = {
			has_core_occupation_modifier = {
				occupied_country_tag = FROM
				modifier = compliance_80
			}
		}

		complete_effect = {
			set_temp_variable = { country_to_initiate = FROM }
			instantiate_collaboration_government = yes
		}

		ai_will_do = {
			factor = 2

			modifier = {
				NOT = { has_decision = SEH_integration_region }
				any_state = {
					is_on_same_continent_as = ROOT
					is_core_of = FROM
					is_core_of = ROOT
				}
				factor = 0
			}

			modifier = {
				tag = MRC
				FROM = { tag = CYR }
				factor = 0
			}

			modifier = {
				tag = MJU
				FROM = {
					OR = {
						tag = CPF
						tag = SPC
					}
				}
				factor = 0
			}

			modifier = {
				# calculating cores we control
				set_temp_variable = { num_cores_we_control = 0 }
				all_of_scopes = {
					array = FROM.core_states

					if = {
						limit = { is_controlled_by = PREV }
						add_to_temp_variable = { num_cores_we_control = 1 }
					}
				}

				# more score if more cores are controlled by us
				add = num_cores_we_control
			}

			# dont make too many collaboratios
			modifier = {
				# as number of collaboration subjects increases
				# only accept subjects with more cores
				set_temp_variable = { num_min_cores = 1.5 }
				all_of_scopes = {
					array = subjects
					if = {
						limit = {
							has_autonomy_state = autonomy_collaboration_government
						}
						add_to_temp_variable = { num_min_cores = 0.7 }
					}
				}

				check_variable = { num_cores_we_control < num_min_cores }

				factor = 0
			}

			# don't create collab governments if you have contested cores
			modifier = {
				any_state = {
					OR = {
						is_controlled_by = ROOT
						is_owned_by = ROOT
					}
					is_core_of = FROM
					is_core_of = ROOT
				}
				factor = 0
			}
		}
	}

	SEH_species_up = {
		icon = generic_prepare_civil_war
		on_map_mode = decision_view_only
		target_array = global.species_decision_states
		state_target = yes

		cost = 1000

		allowed = { always = no }

		target_trigger = {
			any_owned_state = { check_variable = { SEH_species = FROM.SEH_species_decision_target } }
			NOT = {
				is_in_array = { array = SEH_accepted_species_full value = FROM.SEH_species_decision_target }
			}
		}

		available = {
			ROOT = {
				set_temp_variable = { temp_min_proportion = 0.5 }
				set_temp_variable = { temp_species_check = FROM.SEH_species_decision_target }
				count_proportion_owned_species = yes
			}
		}

		complete_effect = {
			set_temp_variable = { SEH_temp_species = FROM.SEH_species_decision_target }
			IF = {
				LIMIT = {
					is_in_array = { array = SEH_accepted_species_partial value = FROM.SEH_species_decision_target }
				}
				SEH_add_accepted_species_full = yes
				ELSE = {
					SEH_add_accepted_species_partial = yes
				}
			}
		}

		ai_will_do = {
			factor = 1
		}
	}
