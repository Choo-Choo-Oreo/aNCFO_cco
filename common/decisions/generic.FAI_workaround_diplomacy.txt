FAI_decision_diplomacy = { # Fantasy AI Diplomacy
	FAI_DD_attache = {
		icon = GFX_decision_generic_decision

		available = {
			# Don't send if already have attaché there
			NOT = { FROM = { has_attache_from = ROOT } }
			
			# FROM must be at war to generate XP
			FROM = { has_war = yes }
			
			# Don't send if we're at war with them
			NOT = { has_war_with = FROM }
			
			# Need sufficient command power for maintenance
			command_power > 49
			
			# Prefer countries we have good relations with
			FROM = {
				has_opinion = { target = ROOT value > 9 }
			}
			OR = {
				has_guaranteed = FROM
				FROM = {
					is_enemy_ideology = no
				}
			}
		}

		visible = {
			# FROM should exist, not be capitulated, and not be a subject
			FROM = {
				exists = yes
				is_subject = no
				has_capitulated = no
			}
		}

		targets_dynamic = yes
		target_array = aligned_to_help_countries
		cost = 100

		days_remove = -1
		
		# Fire only once per target
		fire_only_once = no

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision FAI_DD_attache to [From.GetName]"
			
			diplomatic_relation = {
				country = FROM
				relation = send_attache
				active = yes
			}
			
			FROM = { country_event = FantasyAI_events.1 }
			# Track that we sent attaché for future logic
			set_country_flag = {
				flag = sent_attache_to_@FROM
				days = 365
				value = 1
			}
		}

		ai_will_do = {
			base = 0
			
			### EXPERIENCE GAIN FACTORS ###
			
			# High value: Major wars with lots of divisions
			modifier = {
				add = 250
				FROM = {
					# Check if FROM is in a large-scale war
					any_enemy_country = {
						check_variable = { var = num_divisions value = 50 compare = greater_than }
					}
					# FROM itself has substantial forces
					check_variable = { var = num_divisions value = 40 compare = greater_than }
				}
			}
			
			# Medium value: Moderate conflicts
			modifier = {
				add = 75
				FROM = {
					any_enemy_country = {
						check_variable = { var = num_divisions value = 20 compare = greater_than }
						check_variable = { var = num_divisions value = 50 compare = less_than }
					}
					check_variable = { var = num_divisions value = 20 compare = greater_than }
				}
			}
			
			# Bonus for wars that have been going on (won't end quickly)
			modifier = {
				add = 50
				FROM = {
					any_enemy_country = {
						war_length_with = { tag = FROM months > 3 }
					}
				}
			}
			
			# Bonus if we need specific XP types
			modifier = {
				add = 60
				check_variable = { var = army_experience value = 100 compare = less_than_or_equals }
			}
			
			modifier = {
				add = 80
				check_variable = { var = air_experience value = 100 compare = less_than_or_equals }
				FROM = {
					check_variable = { var = num_deployed_planes value = 500 compare = greater_than }
				}
			}
			
			modifier = {
				add = 80
				check_variable = { var = navy_experience value = 100 compare = less_than_or_equals }
				FROM = {
					check_variable = { var = num_ships value = 30 compare = greater_than }
					any_enemy_country = {
						check_variable = { var = num_ships value = 30 compare = greater_than }
					}
				}
			}
			
			### WAR SUPPORT FACTORS ###
			
			# Critical: Need War Support to change economy laws
			modifier = {
				add = 300
				check_variable = { var = has_war_support value = 25 compare = less_than_or_equals }
				has_idea = civilian_economy
			}
			
			modifier = {
				add = 150
				check_variable = { var = has_war_support value = 35 compare = less_than_or_equals }
				OR = {
					has_idea = civilian_economy
					has_idea = low_economic_mobilisation
				}
			}
			
			# Moderate priority if approaching useful thresholds
			modifier = {
				add = 40
				check_variable = { var = has_war_support value = 10 compare = greater_than_or_equals }
				check_variable = { var = has_war_support value = 20 compare = less_than_or_equals }
			}
			
			modifier = {
				add = 20
				check_variable = { var = has_war_support value = 20 compare = greater_than_or_equals }
				check_variable = { var = has_war_support value = 30 compare = less_than_or_equals }
			}
			
			### INTELLIGENCE FACTORS ###
			
			# High value: Major powers or key regional players
			modifier = {
				add = 80
				FROM = { is_major = yes }
			}
			
			# Strategic neighbors we want to monitor
			modifier = {
				add = 60
				FROM = {
					is_neighbor_of = ROOT
					num_of_military_factories > 20
				}
			}
			
			# Faction members we want full intel on
			modifier = {
				add = 50
				is_in_faction_with = FROM
			}
			
			### NEGATIVE MODIFIERS ###
			
			# Don't spam attachés if we're at war ourselves
			modifier = {
				factor = 0.3
				has_war = yes
				# Unless we desperately need war support
				has_war_support > 0.40
			}
			
			# Reduce if we're low on command power
			modifier = {
				factor = 0.5
				command_power < 20
			}
			
			# Don't send if their war is nearly over
			modifier = {
				factor = 0.1
				FROM = {
					surrender_progress > 0.7
				}
			}
			
			# Reduce if we already sent recently
			modifier = {
				factor = 0.2
				has_country_flag = sent_attache_to_@FROM
			}
			
			# Don't oversaturate with attachés
			modifier = {
				factor = 0.5
				any_other_country = {
					count > 2
					has_attache_from = ROOT
				}
			}
			
			# ROOT has sent volunteers to FROM
			modifier = {
				factor = 0.1
				FROM = {
					any_enemy_country = {
						has_volunteers_amount_from = { tag = PREV.ROOT count > 0 }
					}
				}
			}
			
			# Prefer our faction/allies over others
			modifier = {
				factor = 1.5
				is_in_faction_with = FROM
			}
			
			modifier = {
				factor = 1.3
				has_guaranteed = FROM
			}
			
			# ROOT has sent volunteers to FROM
			modifier = {
				factor = 10.0
				FROM = {
					has_volunteers_amount_from = { tag = PREV.ROOT count > 0 }
				}
			}
			
			# Special: Sino-Japanese War (historical XP farm)
			modifier = {
				add = 120
				OR = {
					FROM = { tag = CHI }
					FROM = { tag = JAP }
				}
				CHI = { has_war_with = JAP }
				is_erly_game =  no
			}
			
			# Special: Spanish Civil War
			modifier = {
				add = 80
				OR = {
					FROM = {
						original_tag = SPR
						is_enemy_ideology = no
					}
				}
				has_global_flag = spanish_civil_war
				is_erly_game =  yes
			}
			
			# Base desire if conditions are met
			modifier = {
				factor = -1
				FROM = { has_war = no }
			}
		}
	}
	# Decision to recall attaché if no longer beneficial
	FAI_DD_recall_attache = {
		icon = GFX_decision_generic_decision
		
		available = {
			# Can always recall
			always = yes
		}
		
		visible = {
			FROM = { has_attache_from = ROOT }
		}
		
		targets_dynamic = yes
		target_array = aligned_to_help_countries
		
		cost = 0
		
		fire_only_once = no
		
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Recalling attaché from [From.GetName]"
			
			diplomatic_relation = {
				country = FROM
				relation = send_attache
				active = no
			}
		}
		
		ai_will_do = {
			base = 0
			
			# Recall if they're no longer at war
			modifier = {
				add = 1000
				FROM = { has_war = no }
			}
			
			# Recall if they capitulated
			modifier = {
				add = 1000
				FROM = { has_capitulated = yes }
			}
			
			# Recall if war is ending soon
			modifier = {
				add = 500
				FROM = {
					any_enemy_country = {
						surrender_progress > 0.9
					}
				}
			}
			
			# Recall if we desperately need command power
			modifier = {
				add = 200
				command_power < 50
				has_war = yes
			}
		}
	}
}
