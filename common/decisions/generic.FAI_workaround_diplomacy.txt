FAI_decision_diplomacy = { # Fantasy AI Diplomacy
	FAI_DD_attache = {
		icon = GFX_decision_generic_decision

		available = {
			# Don't send if already have attaché there
			NOT = { FROM = { has_attache_from = ROOT } }
			
			# FROM must be at war to generate XP
			FROM = { has_war = yes }
			
			# Don't send if we're at war with them
			NOT = { has_war_with = FROM }
			
			# Need sufficient command power for maintenance
			command_power > 49
			
			# Prefer countries we have good relations with
			FROM = {
				has_opinion = { target = ROOT value > 9 }
			}
			OR = {
				has_guaranteed = FROM
				FROM = {
					is_enemy_ideology = no
				}
			}
		}

		visible = {
			# FROM should exist, not be capitulated, and not be a subject
			FROM = {
				exists = yes
				is_subject = no
				has_capitulated = no
			}
		}

		targets_dynamic = yes
		target_array = aligned_to_help_countries
		cost = 100

		days_remove = -1
		
		# Fire only once per target
		fire_only_once = no

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision FAI_DD_attache to [From.GetName]"
			
			diplomatic_relation = {
				country = FROM
				relation = send_attache
				active = yes
			}
			
			FROM = { country_event = FantasyAI_events.1 }
			# Track that we sent attaché for future logic
			set_country_flag = {
				flag = sent_attache_to_@FROM
				days = 365
				value = 1
			}
		}

		ai_will_do = {
			base = 10 # Base desire to start evaluation

			### PEACE-TIME GREED: WAR SUPPORT FOCUS ###
			modifier = {
				add = 500
				has_war = no
				check_variable = { var = has_war_support value = 0.5 compare = less_than }
			}

			modifier = {
				add = 300
				has_idea = civilian_economy
				check_variable = { var = has_war_support value = 0.25 compare = less_than }
			}

			### EXPERIENCE EFFICIENCY ###
			modifier = {
				add = 150
				FROM = {
					any_enemy_country = {
						check_variable = { var = num_divisions value = 50 compare = greater_than }
					}
				}
			}

			modifier = {
				factor = 0.5
				has_army_experience > 250
				has_air_experience > 250
			}

			### STRATEGIC ALIGNMENT ###
			modifier = {
				factor = 10.0
				FROM = {
					has_volunteers_amount_from = { tag = PREV.ROOT count > 0 }
				}
			}

			modifier = {
				add = 120
				OR = { FROM = { tag = CHI } FROM = { tag = JAP } }
				CHI = { has_war_with = JAP }
			}

			### RESOURCE CONSTRAINTS ###
			modifier = {
				factor = 0
				OR = {
					command_power < 10
					AND = {
						has_war = yes
						command_power < 50
					}
				}
			}
		}
	}
	# Decision to recall attaché if no longer beneficial
	FAI_DD_recall_attache = {
		icon = GFX_decision_generic_decision
		
		available = {
			# Can always recall
			always = yes
		}
		
		visible = {
			FROM = { has_attache_from = ROOT }
		}
		
		targets_dynamic = yes
		target_array = aligned_to_help_countries
		
		cost = 0
		
		fire_only_once = no
		
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Recalling attaché from [From.GetName]"
			
			diplomatic_relation = {
				country = FROM
				relation = send_attache
				active = no
			}
		}
		
		ai_will_do = {
			base = 0

			modifier = {
				add = 1000
				OR = {
					FROM = { has_war = no }
					FROM = { has_capitulated = yes }
				}
			}

			# If at peace but WS is maxed, and CP is needed for impending justification
			modifier = {
				add = 200
				has_war = no
				check_variable = { var = has_war_support value = 0.9 compare = greater_than }
				any_neighbor_country = {
					OR = {
						is_justifying_wargoal_against = ROOT
						has_wargoal_against = ROOT
					}
				}
			}

			modifier = {
				add = 500
				FROM = {
					surrender_progress > 0.8
				}
			}

			modifier = {
				add = 300
				has_war = yes
				command_power < 40
				# Only recall if the attaché isn't the primary source of our current WS needs
				check_variable = { var = has_war_support value = 0.5 compare = greater_than }
			}
		}
	}
}
