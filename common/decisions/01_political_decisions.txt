pluralism_on_the_rise = {
	pluralism_switch = {
		icon = generic_prepare_civil_war
		available = {
			pluralism > 0.50
			NOT = { has_government = pluralism }
		}
		visible = {
			or = {
				and = {
					has_idea_with_trait = pluralism_advisor
					NOT = { has_government = pluralism }
				}
				and = {
					pluralism > 0.45
					NOT = { 
						has_government = pluralism
						has_idea_with_trait = pluralism_advisor
					}
				}
			}
		}

			country_event = { id = decision_events.1 }
		}
		ai_will_do = {
			base = 5
			modifier = {
				add = -9999
				pre_civilwar_trigger = yes
			}
		}
	}
}
