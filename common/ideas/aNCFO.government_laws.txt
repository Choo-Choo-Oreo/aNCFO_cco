				    if = { limit = { check_variable = { var = denom_aluminium value = 0 compare = greater_than } } divide_temp_variable = { ratio_aluminium = denom_aluminium } }

				    set_temp_variable = { ratio_rubber = resource_produced@rubber }
				    set_temp_variable = { denom_rubber = resource_consumed@rubber }
				    if = { limit = { check_variable = { var = denom_rubber value = 0 compare = equals } } set_temp_variable = { ratio_rubber = 999999 } }
				    if = { limit = { check_variable = { var = denom_rubber value = 0 compare = greater_than } } divide_temp_variable = { ratio_rubber = denom_rubber } }

				    set_temp_variable = { ratio_tungsten = resource_produced@tungsten }
				    set_temp_variable = { denom_tungsten = resource_consumed@tungsten }
				    if = { limit = { check_variable = { var = denom_tungsten value = 0 compare = equals } } set_temp_variable = { ratio_tungsten = 999999 } }
				    if = { limit = { check_variable = { var = denom_tungsten value = 0 compare = greater_than } } divide_temp_variable = { ratio_tungsten = denom_tungsten } }

				    set_temp_variable = { ratio_steel = resource_produced@steel }
				    set_temp_variable = { denom_steel = resource_consumed@steel }
				    if = { limit = { check_variable = { var = denom_steel value = 0 compare = equals } } set_temp_variable = { ratio_steel = 999999 } }
				    if = { limit = { check_variable = { var = denom_steel value = 0 compare = greater_than } } divide_temp_variable = { ratio_steel = denom_steel } }

				    set_temp_variable = { ratio_chromium = resource_produced@chromium }
				    set_temp_variable = { denom_chromium = resource_consumed@chromium }
				    if = { limit = { check_variable = { var = denom_chromium value = 0 compare = equals } } set_temp_variable = { ratio_chromium = 999999 } }
				    if = { limit = { check_variable = { var = denom_chromium value = 0 compare = greater_than } } divide_temp_variable = { ratio_chromium = denom_chromium } }

				    set_temp_variable = { temp = 1.0 }

				    # Strong boost if any critical resource is in deficit
				    if = {
				        limit = {
				            OR = {
				                check_variable = { var = ratio_oil value = 1 compare = less_than }
				                check_variable = { var = ratio_steel value = 1 compare = less_than }
				                check_variable = { var = ratio_coal value = 1 compare = less_than }
				                check_variable = { var = ratio_tungsten value = 1 compare = less_than }
				            }
				        }
				        multiply_temp_variable = { temp = 2.50 }   # aggressively choose closed when critical deficits present
				    }

				    # Boost if importing several resources (import-dependency)
				    if = {
				        limit = {
				            AND = {
				                OR = {
				                    check_variable = { var = resource_imported@oil value = 0 compare = greater_than }
				                    check_variable = { var = resource_imported@coal value = 0 compare = greater_than }
				                    check_variable = { var = resource_imported@steel value = 0 compare = greater_than }
				                }
				                OR = {
				                    check_variable = { var = resource_imported@aluminium value = 0 compare = greater_than }
				                    check_variable = { var = resource_imported@tungsten value = 0 compare = greater_than }
				                    check_variable = { var = resource_imported@chromium value = 0 compare = greater_than }
				                }
				            }
				        }
				        multiply_temp_variable = { temp = 1.8 }   # multiple import signals → prefer closed
				    }

				    # Penalty if broad surpluses across key resources (do not close if you have plenty)
				    if = {
				        limit = {
				            AND = {
				                check_variable = { var = ratio_oil value = 2 compare = greater_than }
				                check_variable = { var = ratio_steel value = 2 compare = greater_than }
				                check_variable = { var = ratio_coal value = 2 compare = greater_than }
				            }
				        }
				        multiply_temp_variable = { temp = 0.10 }   # very unlikely to choose closed when many surpluses
				    }

				    factor = temp
				}
   			}
		}
	}
